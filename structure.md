/etc/spectrumos/              # Default configs (editable)

├── waybar/                   # Default Waybar configs
├── spectrumos.conf           # Main SpectrumOS settings
└── wallpaper.conf            # Wallpaper/pywal preferences

├── current.jpg               # Current wallpaper (for SDDM/Limine)

├── spectrumos.conf           # User preferences
└── wallpaper.conf            # Personal wallpaper settings
